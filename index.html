        const countCharacters = (S) => {
  let countA = 0, countD = 0;
  for(let i = 0; i < S.length; i++){
    if(S[i] === 'A') countA++;
    else if(S[i] === 'D') countD++;
  }
  const arr = [countA, countD];
  return arr;
};

Question2. Count the Heads

Answer.

        var Count_Occ = (s) => {
  
const countHead = {};
// console.log(s.split(''), s.split('').sort(), s.split('').sort().join(''));
const so = s.split('').sort().join('');
let char, i, count, str = '';

for (i = 0; i < so.length; ++i) {
  char = so.charAt(i);
  count = countHead[char];
  countHead[char] = count ? count + 1: 1;
}
for (char in countHead) {
  if(char!=' ' && countHead[char] > 1)
    str += char + countHead[char];
}
return str;
};

Question6. Find the Winner

Answer.

        var Game_Winner = (S) => {
  let countA = 0, countD = 0;
  for(let i = 0; i < S.length; i++){
    if(S[i] === 'A') countA++;
    else if(S[i] === 'D') countD++;
  }
  if(countA > countD) return 'Aditya';
  else if(countA < countD) return 'Danish';
  return 'Draw';
}
